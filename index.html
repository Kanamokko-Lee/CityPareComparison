// --- 1. 基本となる国内主要フィックス（キャッシュ用） --- // ここにないポイントは、下の fetchCoordinatesFromServer() を通じて自動で外部から取得・補完されます let FIX_DB = { "NRT": [35.7767, 140.3864], "SGN": [10.8188, 106.6519], "SAMUS": [34.7833, 137.9500], "PIGOK": [34.2000, 137.1000] }; // --- 2. 外部航空データAPIなどから緯度経度をリアルタイム検索する関数 --- async function fetchCoordinatesFromServer(pointName) { // すでにローカルDBにある場合はそれを返す if (FIX_DB[pointName]) return FIX_DB[pointName]; try { // 全世界の航空データをインデックスしているオープンAPI（一例としてgeojson.xyzやOurAirportsベースのミラー） // ※デモ用に信頼性の高いオープンAPIプロキシや、自身でホストしたJSONへリクエストを送る想定です const response = await fetch(`https://api.navdata.io/v1/fix/${pointName}`, { method: 'GET', headers: { 'Accept': 'application/json' } }); if (response.ok) { const data = await response.json(); if (data && data.latitude && data.longitude) { // 取得成功したら次回のためにキャッシュに保存 FIX_DB[pointName] = [data.latitude, data.longitude]; return FIX_DB[pointName]; } } } catch (error) { console.warn(`API検索失敗 (${pointName}):`, error); } // APIで見つからない、またはエラー時のフォールバック対策 // （前後のポイントから推定するか、今回はスキップさせるためにnullを返す） return null; } // --- 3. 解析・プロット処理（非同期通信対応版） --- async function parseAndPlot() { const text = document.getElementById('textInput').value; const statusDiv = document.getElementById('status'); const legendBody = document.getElementById('legendBody'); routeLayers.forEach(l => map.removeLayer(l)); routeLayers = []; legendBody.innerHTML = ''; if (!text.trim()) return; const variants = text.split(/VRNT\s+/).slice(1); if (variants.length === 0) return; statusDiv.innerText = "ウェイポイントの座標を検索・計算中..."; // 各バリアントを非同期で順次処理 for (let index = 0; index < variants.length; index++) { const vStr = variants[index]; const lines = vStr.split('\n').map(l => l.trim()).filter(l => l); // ルート名・TRIPF・EETの抽出（前述と同様） let routeName = `VRNT ${index+1}`; const viaMatch = vStr.match(/VIA\s+LOZ-?([A-Z0-9]+)/i) || vStr.match(/VIA\s+([A-Z0-9]+)/i); if (viaMatch) routeName = viaMatch[1]; const tripfMatch = vStr.match(/TRIPF\s+(\d+)/i); const eetMatch = vStr.match(/EET\s+([\d\.]+)/i); const tripf = tripfMatch ? parseInt(tripfMatch[1]).toLocaleString() + " lbs" : "N/A"; const eet = eetMatch ? eetMatch[1] : "N/A"; const color = ROUTE_COLORS[index % ROUTE_COLORS.length]; const latlngs = [FIX_DB.NRT]; // 成田スタート // テキストから単語（Fix候補）を抽出 const words = vStr.replace(/[\/\(\)]/g, ' ').split(/\s+/); for (const word of words) { const cleanWord = word.toUpperCase().trim(); // 2文字以上5文字以下の大文字英数字（航空路名やFixの標準）を対象 if (cleanWord.length >= 2 && cleanWord.length <= 5 && cleanWord !== "NRT" && cleanWord !== "SGN") { // API通信を含めて座標を取得 const coords = await fetchCoordinatesFromServer(cleanWord); if (coords) { if (latlngs[latlngs.length - 1] !== coords) { latlngs.push(coords); } } } } latlngs.push(FIX_DB.SGN); // ホーチミンゴール // ポリライン描画 const polyline = L.polyline(latlngs, { color: color, weight: 4, opacity: 0.8 }).addTo(map); polyline.bindPopup(`Route: ${routeName}
TRIPF: ${tripf}
EET: ${eet}`); routeLayers.push(polyline); // 凡例追加 const tr = document.createElement('tr'); tr.innerHTML = ` ${routeName}${tripf}${eet}`; legendBody.appendChild(tr); } if(routeLayers.length > 0) { const group = new L.FeatureGroup(routeLayers); map.fitBounds(group.getBounds().pad(0.1)); statusDiv.innerText = "すべてのウェイポイントを取得し、プロットが完了しました。"; } }
